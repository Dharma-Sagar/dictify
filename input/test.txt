རུས་སྦལ་ ལོང་བ་ -འི་ སྒྲུང་ ། སྔོན་མ་ སྔོན་མ་ ཞིག་ ལ་ རྒྱ་མཚོ་ ཆེན་པོ་ ཞིག་ ཡོད་པ་རེད་ ། གྲུ་གཟིངས་ ནང་ དུ་ ཟླ་བ་ མང་པོ་ -ར་ སོང་ ཡང་ རྒྱ་མཚོ་ དེ་ བརྒལ་ ཐུབ་ ཀྱི་ ཡོད་མ་རེད་ ། རྒྱ་མཚོ་ དེ་ -འི་ འོག་ ལ་ མིག་ གཉིས་ཀ་ ལོང་བ་ ཡིན་པ་ -འི་ རུས་སྦལ་ ཆེན་པོ་ ཞིག་ ཡོད་པ་རེད་ ། རུས་སྦལ་ དེ་ -ས་ རྒྱ་མཚོ་ -འི་ ནང་ དུ་ སྐྱེས་པ་ -འི་ སྔོ་ཚལ་ གྱི་ རིགས་ མང་པོ་ ཟོས་ ནས་ ལོ་ བརྒྱ་ ལྷག་ གི་ རིང་ ལ་ གཉིད་ ཁུག་ གི་ཡོད་རེད་ ། ལོ་ བརྒྱ་ རེ་ -འི་ མཚམས་ སུ་ མགོ་ ཡར་ བཀྱག་ ནས་ རྒྱ་མཚོ་ -འི་ ཕྱི་ལོགས་ ལ་ ཐེངས་མ་ གཅིག་ ལྟ་ ཡི་ཡོད་པ་རེད་ ། དེ་ནས་ སྔོ་ཚལ་ གྲོད་ཁོག་ གང་ ཟོས་ རྗེས་ ཡང་བསྐྱར་ གཉིད་ ཁུག་ ནས་ བསྡད་ ཀྱི་ཡོད་ ། རྒྱ་མཚོ་ དེ་ ལ་ རྦ་རླབས་ ཤུགས་ ཆེན་པོ་ ཡོད་རེད་ ། རྒྱ་མཚོ་ དེ་ -འི་ ཁ་ ན་ ། ཤིང་ལེབ་ དཀྱིལ་མ་ ལ་ ཨི་ཁུང་ ཆེ་ཆུང་ རུས་སྦལ་ གྱི་ མགོ་ དང་ འདྲ་པོ་ ཞིག་ ཡོད་པ་རེད་ ། ཤིང་ལེབ་ དེ་ རྒྱ་མཚོ་ -འི་ རྦ་རླབས་ ཀྱིས་ ཤར་ ནས་ ནུབ་ དང་ ། བྱང་ ནས་ ལྷོ་ ལ་ འཁྱེར་ གྱི་ ཡོད་ རེད་ ། ལྷ་ ཞིག་ གིས་ རུས་སྦལ་ དེ་ ལ་ ། ཁྱོད་རང་ གི་ མགོ་ ཤིང་ལེབ་ ཀྱི་ ཨི་ཁུང་ ནང་ ལ་ ག་དུས་ བཅུག་ ཐུབ་ ན་ ། ཁྱེད་རང་ གི་ མིག་ གཉིས་ ཀྱིས་ མཐོང་ ཐུབ་ ཀྱི་རེད་ ཅེས་ གསུངས་ པ་རེད་ ། རུས་སྦལ་ གྱི་ སེམས་ ལ་ ཡང་ ། ང་ -འི་ མགོ་ ཤིང་ལེབ་ ཀྱི་ ཨི་ཁུང་ ནང་ བཅུག་ ཐུབ་ ན་ ཡག་པོ་ འདུག་ བསམ་ པ་རེད་ ། ཡིན་ན་ཡང་ ། རུས་སྦལ་ གྱི་ མིག་ གིས་ མཐོང་ ཐུབ་ ཀྱི་ཡོད་མ་རེད་ ། ཁ་ལག་ རྒྱགས་ ཚད་ ཟོས་ རྗེས་ གཉིད་ ཁུག་ གི་ཡོད་རེད་ ། ཤིང་ལེབ་ དེ་ འཚོལ་ དུ་ འགྲོ་ ཤེས་ ཀྱི་མེད་པ་ མ་ཟད་ ། ཤིང་ལེབ་ དེ་ ག་པ་ -ར་ ཡོད་ མེད་ ཀྱང་ ཤེས་ ཀྱི་ཡོད་མ་རེད་ །
